    class SomeKindOfClass

    end

Bad

    class someOtherKindOfClass

    end

    class foo_bar

    end

## Method

Good

    def foo_bar_something(foo, bar)
      ...
    end

    def foo_bar_something foo, bar
      ...
    end

Bad

    def badMethodName( foo, bar )
      ...
    end
